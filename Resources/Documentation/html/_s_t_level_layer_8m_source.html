190 - (void)updateCollisions:(ccTime)delta {
191  for (STGameObject *child in self.gameObjects) {
192  for (STGameObject *child2 in self.gameObjects) {
193  // Don't check same object
194  if (child == child2) {
195  break;
196  }
197 
198  if (STRectIntersect(child.boundingBox, child2.boundingBox)) {
199  // Position objects
200  STRectEdge edge1 = [self updateCollisionOfGameObject:child withGameObject:child2 delta:delta];
201  STRectEdge edge2 = [self updateCollisionOfGameObject:child2 withGameObject:child delta:delta];
202 
203  // Send notifications
204  [child collisionWithGameObject:child2 edge:edge1];
205  [child2 collisionWithGameObject:child edge:edge2];
206  }
207  }
208  }
209 }

211 - (STRectEdge)updateCollisionOfGameObject:(STGameObject *)gameObject
212  withGameObject:(STGameObject *)gameObject2
213  delta:(ccTime)delta {
214 
215  STRectEdge rectEdge;
216  float edgeLeft = (gameObject.boundingBox.origin.x - gameObject2.boundingBox.origin.x - gameObject.boundingBox.size.width) * -1;
217  float edgeRight = (gameObject.boundingBox.origin.x + gameObject2.boundingBox.size.width - gameObject2.boundingBox.origin.x);
218  float edgeTop = (gameObject.boundingBox.origin.y + gameObject.boundingBox.size.height - gameObject2.boundingBox.origin.y);
219  float edgeBottom = (gameObject.boundingBox.origin.y - gameObject2.boundingBox.size.height - gameObject2.boundingBox.origin.y) * -1;
220 
221  float offset = 0.0;
222  if (edgeLeft < edgeRight) {
223  rectEdge = STRectEdgeMinX;
224  offset = edgeLeft;
225  } else {
226  rectEdge = STRectEdgeMaxX;
227  offset = edgeRight;
228  }
229 
230  if (edgeTop < edgeBottom) {
231  float cached = edgeTop;
232  if (cached < offset) {
233  rectEdge = STRectEdgeMaxY;
234  offset = cached;
235  }
236  } else {
237  float cached = edgeBottom;
238  if (cached < offset) {
239  rectEdge = STRectEdgeMinY;
240  offset = cached;
241  }
242  }
243 
244  if (gameObject.bodyType != STGameObjectBodyTypeStatic) {
245  if (gameObject2.bodyType != STGameObjectBodyTypeStatic) {
246  offset /= 2.0;
247  }
248 
249  if ([gameObject bodyType] != STGameObjectBodyTypeNonColliding && [gameObject2 bodyType] != STGameObjectBodyTypeNonColliding) {
250  switch (rectEdge) {
251  case STRectEdgeMinX:
252  {
253  [gameObject move:ccp(offset, 0)];
254  }
255  break;
256  case STRectEdgeMaxX:
257  {
258  [gameObject move:ccp(-offset, 0)];
259  }
260  break;
261  case STRectEdgeMinY:
262  {
263  [gameObject move:ccp(0, offset)];
264  }
265  break;
266  case STRectEdgeMaxY:
267  {
268  [gameObject move:ccp(0, -offset)];
269  }
270  break;
271  }
272 
273  if (rectEdge == STRectEdgeMinY && gameObject.velocity.y < 0) {
274  gameObject.velocity = ccp(gameObject.velocity.x, 0);
275  }
276  if (rectEdge == STRectEdgeMaxY && gameObject.velocity.y > 0) {
277  gameObject.velocity = ccp(gameObject.velocity.x, 0);
278  }
279  }
280  }
281 
282  return rectEdge;
283 }

289 #define kMapLoadingDelta 10.0
290 - (void)readGameObjectsFromMap {
291  int mapHeight = self.objectLayer.layerSize.height;
292  CGFloat cameraWidth = [[CCDirector sharedDirector] winSize].width;
293  CGFloat mapX = self.map.position.x * -1 / self.map.scale;
294 
295  int tileWidth = (cameraWidth + (2 * kMapLoadingDelta)) / self.map.tileSize.width;
296  int firstTile = ((mapX - kMapLoadingDelta) / self.map.tileSize.width) + 1;
297  if (firstTile < 0) firstTile = 0;
298 
299  int lastTile = firstTile + tileWidth;
300  if (lastTile > self.objectLayer.layerSize.width) lastTile = self.objectLayer.layerSize.width;
301 
302  for (int x = firstTile; x < lastTile; x++) {
303  for (int y = 0; y < mapHeight; y++) {
304  [self createGameObjectAtX:x y:y];
305  }
306  }
307 }

310 - (void)trashLostGameObjects {
311  CGFloat cameraWidth = [[CCDirector sharedDirector] winSize].width;
312  CGFloat mapX = self.map.position.x * -1 / self.map.scale;
313 
314  for (STGameObject *gameObject in self.gameObjects) {
315  if (
316  // if objects fell of the ground
317  gameObject.boundingBox.origin.y + gameObject.boundingBox.size.height + kRemovingDelta < 0
318  ||
319  // too much on left side
320  gameObject.boundingBox.origin.x + gameObject.boundingBox.size.width + kRemovingDelta < mapX
321  ||
322  // too much on right side
323  gameObject.boundingBox.origin.x - kRemovingDelta > mapX + cameraWidth
324  ) {
325  [gameObject setDead:YES];
326  }
327  }

383 - (void)player:(STPlayer *)player didMoveToPoint:(CGPoint)point {
384  CGFloat cameraWidth = [[CCDirector sharedDirector] winSize].width;
385  CGFloat playerX = player.position.x;
386  CGFloat mapX = self.map.position.x * -1 / self.map.scale;
387 
388  // Moves the camera with the player
389  if (mapX - playerX + kMaxCameraEdge < 0) {
390  float newMapX = (-playerX + kMaxCameraEdge) * self.map.scale;
391  if (newMapX > 0) {
392  newMapX = 0;
393  } else if (self.map.boundingBox.size.width + newMapX - cameraWidth < 0) {
394  newMapX = -self.map.boundingBox.size.width + cameraWidth;
395  }
396 
397  self.map.position = ccp(newMapX, self.map.position.y);;
398  }
