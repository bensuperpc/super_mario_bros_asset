79 -(void)updateVelocity:(CGPoint)point
80 {
81  // Calculate distance and angle from the center.
82  float dx = point.x;
83  float dy = point.y;
84  float dSq = dx * dx + dy * dy;
85 
86  if(dSq <= deadRadiusSq){
87  velocity = CGPointZero;
88  degrees = 0.0f;
89  stickPosition = point;
90  return;
91  }
92 
93  float angle = atan2f(dy, dx); // in radians
94  if(angle < 0){
95  angle += SJ_PI_X_2;
96  }
97  float cosAngle;
98  float sinAngle;
99 
100  if(isDPad){
101  float anglePerSector = 360.0f / numberOfDirections * SJ_DEG2RAD;
102  angle = roundf(angle/anglePerSector) * anglePerSector;
103  }
104 
105  cosAngle = cosf(angle);
106  sinAngle = sinf(angle);
107 
108  // NOTE: Velocity goes from -1.0 to 1.0.
109  if (dSq > joystickRadiusSq || isDPad) {
110  dx = cosAngle * joystickRadius;
111  dy = sinAngle * joystickRadius;
112  }
113 
114  velocity = CGPointMake(dx/joystickRadius, dy/joystickRadius);
115  degrees = angle * SJ_RAD2DEG;
116 
117  // Update the thumb's position
118  stickPosition = ccp(dx, dy);
119 }
